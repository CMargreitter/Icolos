[docs]def add_citation(
    step_type: str,
    workflow_step_types: List[str],
    citations: List[str],
    citation_string: str,
):
    if (
        any(
            [
                True
                for w_step_type in workflow_step_types
                if step_type.upper() in w_step_type
            ]
        )
        or step_type == "default"
    ):
        citations.append(citation_string)
